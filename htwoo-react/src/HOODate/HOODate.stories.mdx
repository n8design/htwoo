## With Min/Max values set
<Canvas>
  <Story 
    name="Min/Max Values"
    args = {
      (function () {
        const _minValue = new Date();
        const _maxValue = new Date();

        _maxValue.setFullYear(_maxValue.getFullYear() + 1);
        _minValue.setFullYear(_minValue.getFullYear() - 1);

        return { value: (new Date()).toLocaleDateString(), minValue: _minValue.toISOString().split('T')[0], maxValue: _maxValue.toISOString().split('T')[0] };
      })()
    }>
    {Template.bind({})}
  </Story>
</Canvas>
