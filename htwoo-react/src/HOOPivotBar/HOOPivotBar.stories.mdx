# HOOPivotBar

The `options` property requires an array of items of type `IHOOPivotItem`.

```ts
export interface IHOOPivotItem {
  key: number | string;
  text: string;
}
```

## Arguments

<ArgTypes of={HOOPivotBar} />

export const options = [{ key: 1, text: "Menu 1" }, { key: 2, text: "Menu 2" }, { key: 3, text: "Menu 3" }, { key: 4, text: "Menu 4" }, { key: 5, text: "Menu 5" }];
export const Template = (args) => {
  const [_, updateArgs] = useArgs();

  const handle = (ev, key) => {
    args.selectedKey = key;
    updateArgs({ ...args });
  };

  return <HOOPivotBar {...args} onClick={handle} />;
};
